 private client: Opik;

constructor() {  
 this.client = new Opik({  
 apiKey: process.env.OPIK_API_KEY,  
 apiUrl: process.env.OPIK_URL_OVERRIDE,  
 projectName: 'ikpa-financial-coach',  
 workspaceName: process.env.OPIK_WORKSPACE_NAME,  
 });  
 }

async traceConversation(  
 userId: string,  
 input: string,  
 output: string,  
 metadata: Record<string, any>  
 ) {  
 const trace = this.client.trace({  
 name: 'financial-coach-conversation',  
 input: { userId, prompt: input },  
 output: { response: output },  
 metadata: {  
 ...metadata,  
 cashFlowScore: metadata.cashFlowScore,  
 hasDebts: metadata.hasDebts,  
 familyObligations: metadata.familyObligations,  
 },  
 });

     // Add span for Claude call
     const span = trace.span({
       name: 'claude-completion',
       type: 'llm',
       input: { prompt: input },
       output: { completion: output },
     });

     span.end();
     trace.end();
     await this.client.flush();

     return trace;

}  
 }
